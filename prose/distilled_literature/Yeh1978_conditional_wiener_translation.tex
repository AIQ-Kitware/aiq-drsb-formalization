\section*{Source map and scope}
\begin{sourceguide}
\item[Primary source] J.~Yeh, \emph{Transformation of Conditional Wiener Integrals under Translation and the Cameron--Martin Translation Theorem}, \emph{Tohoku Mathematical Journal} 30 (1978), 505--515 \cite{Yeh1978}.
\item[Coverage] The Wiener-space setup in Section 1; Section 2's definition, Lemmas 1--2, and Theorem 1 with equations (2.1)--(2.19); Section 3's translation setup, Cameron--Martin input, Theorem 2, the corollary, and the concluding exponential-integral example through equation (3.12) and the endpoint-restricted integral.
\item[Method] Yeh's order is preserved.  First transform the conditioning map, then prove a finite-measure change-of-variables lemma, combine the two through test functions, and only afterward specialize to Wiener translation and insert the Gaussian endpoint density ratio.
\item[Supplemental references] The translation density used by Yeh is the Cameron--Martin theorem, with the $L^2$ extension attributed in the source to Sunouchi and Segal and the stochastic integral interpreted through Paley--Wiener \cite{CameronMartin1944Translations,Sunouchi1951,Segal1958,PaleyWiener1934}.  These are cited inputs, not arguments proved in Yeh's article.
\end{sourceguide}

\section{Wiener measure and conditional Wiener integrals}
\sourceanchor{Section 1, equations (1.1)--(1.6).}
Fix $t\in(0,\infty)$ and let
\[
 C_0[0,t]:=\{x\in C[0,t]:x(0)=0\}.
\]
For times $0=s_0<s_1<\cdots<s_n\le t$ and Borel $B\subset\R^n$, a Wiener cylinder is
\[
 W=\{x\in C_0[0,t]:(x(s_1),\ldots,x(s_n))\in B\}.
\]
Yeh recalls the usual finite-dimensional Gaussian formula
\begin{align}
 m_w(W)
 &=(2\pi)^{-n/2}\prod_{j=1}^n(s_j-s_{j-1})^{-1/2}\notag\\
 &\quad\times
 \int_B
 \exp\left
 \{-\frac12\sum_{j=1}^n
 \frac{(\xi_j-\xi_{j-1})^2}{s_j-s_{j-1}}
 \right\}
 d\xi_1\cdots d\xi_n,
 \qquad \xi_0=0.\label{eq:Yeh-cylinder}
\end{align}
The completed Caratheodory extension is again denoted $m_w$.

The conditioning functional throughout the paper is the endpoint
\begin{equation}\label{eq:Yeh-endpoint}
 X[x]:=x(t).
\end{equation}
Its law is
\begin{equation}\label{eq:Yeh-endpoint-law}
 P_X(d\xi)
 =\frac1{\sqrt{2\pi t}}e^{-\xi^2/(2t)}d\xi.
\end{equation}
A conditional Wiener integral $E^w(Y\mid X)$ is not regarded merely as a random variable on path space: it is an equivalence class of measurable functions of the endpoint variable on $(\R,\mathcal B(\R),P_X)$.  Yeh uses the same notation for the class and for an arbitrary version.

\section{Conditional expectations indexed by a measurable map}
\sourceanchor{Definition in Section 2, equations (2.1)--(2.3).}
Let $(\Omega,\mathcal B,P)$ be a probability space, let $(S,\mathcal S)$ be measurable, and let $X:\Omega\to S$ be measurable.  Denote the pushforward by
\[
 P_X(F):=P(X^{-1}F).
\]
For an integrable real random variable $Z$, a version of $E(Z\mid X)$ is an $\mathcal S$-measurable, $P_X$-integrable function $\Psi:S\to\R$ such that
\begin{equation}\label{eq:Yeh-definition}
 \int_{X^{-1}(F)}Z\,dP
 =\int_F\Psi(\xi)P_X(d\xi)
 \qquad(F\in\mathcal S).
\end{equation}
Equivalently, for every measurable test function $g$ for which one side exists,
\begin{equation}\label{eq:Yeh-test}
 E[(g\circ X)Z]
 =\int_Sg(\xi)E(Z\mid X)(\xi)P_X(d\xi).
\end{equation}
Yeh repeatedly uses the ``restricted sense'': an identity involving possibly nonintegrable products means that existence of either side implies existence and equality of the other.

\section{Lemma 1: transformation of the conditioning map}
\sourceanchor{Lemma 1, equations (2.4)--(2.10).}
Let $T:\Omega\to\Omega$ be measurable and let $h:S\to S$ be a measurable bijection with measurable inverse.  Assume
\begin{equation}\label{eq:Yeh-commutation}
 X\circ T=h\circ X
 \qquad P\text{-a.s.}
\end{equation}

\begin{lemma}[Yeh, Lemma 1]\label{lem:Yeh-1}
For arbitrary versions,
\begin{equation}\label{eq:Yeh-lemma1}
 E(Z\mid X\circ T)(\xi)
 =E(Z\mid X)(h^{-1}\xi)
 \quad P_{X\circ T}\text{-a.e.}
\end{equation}
\end{lemma}

\begin{proof}
Fix $F\in\mathcal S$.  The almost-sure identity \eqref{eq:Yeh-commutation} implies that
\[
 (X\circ T)^{-1}(F)
 \quad\text{and}\quad
 (h\circ X)^{-1}(F)=X^{-1}(h^{-1}F)
\]
have null symmetric difference.  Hence
\begin{align}
 \int_FE(Z\mid X\circ T)(\xi)P_{X\circ T}(d\xi)
 &=\int_{X^{-1}(h^{-1}F)}Z\,dP\notag\\
 &=\int_{h^{-1}F}E(Z\mid X)(\eta)P_X(d\eta).\label{eq:Yeh-L1-middle}
\end{align}
Now push $P_X$ forward by $h$.  If $P_h=h_\#P_X$, then
\begin{align*}
 \int_{h^{-1}F}E(Z\mid X)(\eta)P_X(d\eta)
 &=\int_FE(Z\mid X)(h^{-1}\xi)P_h(d\xi).
\end{align*}
But
\[
 P_h(F)=P_X(h^{-1}F)=P((h\circ X)^{-1}F)=P_{X\circ T}(F),
\]
where the last equality again uses \eqref{eq:Yeh-commutation}.  Substitution into \eqref{eq:Yeh-L1-middle} and uniqueness of Radon--Nikodym representatives proves \eqref{eq:Yeh-lemma1}.
\end{proof}

\section{Lemma 2: transformation of a finite measure}
\sourceanchor{Lemma 2, equations (2.11)--(2.12).}
Let $(\Omega,\mathcal B,\mu)$ be a finite measure space and let $T:\Omega\to\Omega$ be measurable.  Suppose there is a measurable real function $J$ satisfying
\begin{equation}\label{eq:Yeh-density}
 \mu(A)=\int_{T^{-1}A}J(\omega)\mu(d\omega)
 \qquad(A\in\mathcal B).
\end{equation}

\begin{lemma}[Yeh, Lemma 2]\label{lem:Yeh-2}
For every measurable real function $Z$,
\begin{equation}\label{eq:Yeh-change}
 \int_\Omega Z(\omega)\mu(d\omega)
 =\int_\Omega Z(T\omega)J(\omega)\mu(d\omega)
\end{equation}
in the restricted sense.
\end{lemma}

\begin{proof}
For a simple function $Z=\sum_{j=1}^nc_j\one_{A_j}$,
\begin{align*}
 \int Z\,d\mu
 &=\sum_jc_j\mu(A_j)
 =\sum_jc_j\int_{T^{-1}A_j}J\,d\mu\\
 &=\int\sum_jc_j\one_{A_j}(T\omega)J(\omega)\mu(d\omega)
 =\int(Z\circ T)J\,d\mu.
\end{align*}
For $Z\ge0$, approximate increasingly by simple functions and apply monotone convergence.  For a real-valued $Z$, apply the nonnegative result to $Z^+$ and $Z^-$; the restricted-sense convention handles the cases in which only one of the two signed integrals is initially known to exist.
\end{proof}

\section{The abstract conditional transformation theorem}
\sourceanchor{Theorem 1, conditions $1^\circ$--$3^\circ$, equations (2.13)--(2.19).}
Assume
\begin{align}
 P(A)&=\int_{T^{-1}A}J\,dP,
 &&A\in\mathcal B,\label{eq:Yeh-T1-1}\\
 X\circ T&=h\circ X,
 &&P\text{-a.s.},\label{eq:Yeh-T1-2}
\end{align}
where $h$ is a bimeasurable bijection.

\begin{theorem}[Yeh, Theorem 1]\label{thm:Yeh-1}
For arbitrary versions and every measurable test function $g$,
\begin{align}
 &\int_Sg(\xi)E(Z\mid X)(\xi)P_X(d\xi)\notag\\
 &\qquad=
 \int_Sg(\xi)
 E((Z\circ T)J\mid X)(h^{-1}\xi)
 P_{X\circ T}(d\xi),\label{eq:Yeh-T1-integral}
\end{align}
in the restricted sense.  If additionally
\begin{equation}\label{eq:Yeh-AC}
 P_{X\circ T}\ll P_X,
\end{equation}
then
\begin{equation}\label{eq:Yeh-T1-pointwise}
 E(Z\mid X)(\xi)
 =E((Z\circ T)J\mid X)(h^{-1}\xi)
 \frac{dP_{X\circ T}}{dP_X}(\xi)
 \quad P_X\text{-a.e.}
\end{equation}
\end{theorem}

\begin{proof}
Start from \eqref{eq:Yeh-test}:
\[
 \int_Sg(\xi)E(Z\mid X)(\xi)P_X(d\xi)
 =E[(g\circ X)Z].
\]
Lemma~\ref{lem:Yeh-2}, applied to $(g\circ X)Z$, gives
\begin{align*}
 E[(g\circ X)Z]
 &=E[(g\circ X\circ T)(Z\circ T)J].
\end{align*}
Condition first on $X\circ T$:
\begin{align*}
 E[(g\circ X\circ T)(Z\circ T)J]
 &=\int_Sg(\xi)
 E((Z\circ T)J\mid X\circ T)(\xi)
 P_{X\circ T}(d\xi).
\end{align*}
Lemma~\ref{lem:Yeh-1} replaces the conditional expectation by
\[
 E((Z\circ T)J\mid X)(h^{-1}\xi),
\]
which proves \eqref{eq:Yeh-T1-integral}.  Under \eqref{eq:Yeh-AC}, rewrite the right side against $P_X$ and choose $g=\one_F$ for arbitrary $F\in\mathcal S$.  Equality of the resulting integrals over every $F$ identifies the integrands and yields \eqref{eq:Yeh-T1-pointwise}.
\end{proof}

\section{The Cameron--Martin input used by Yeh}
\sourceanchor{Section 3, equations (3.1)--(3.4).}
Fix $x_0\in C_0[0,t]$ and define the path translation
\begin{equation}\label{eq:Yeh-T}
 T[x]:=x+x_0.
\end{equation}
For differentiable $x_0$ with $x_0'$ of bounded variation, the Cameron--Martin translation theorem gives measurability of translated Wiener sets and
\begin{equation}\label{eq:Yeh-CM}
 m_w(\Gamma)
 =\int_{T^{-1}\Gamma}J(x)m_w(dx),
\end{equation}
where
\begin{equation}\label{eq:Yeh-J}
 J(x)
 =\exp\left\{-\frac12\int_0^t|x_0'(s)|^2ds\right\}
  \exp\left\{-\int_0^tx_0'(s)\,dx(s)\right\}.
\end{equation}
The second integral is Riemann--Stieltjes in this first regularity class.  Yeh then invokes Sunouchi and Segal to extend \eqref{eq:Yeh-CM} to absolutely continuous $x_0$ with $x_0'\in L^2[0,t]$, interpreting the second integral as a Paley--Wiener--Zygmund--Riemann--Stieltjes integral.  It exists outside a Wiener-null set, on which $J$ may be defined arbitrarily because the completed Wiener space is complete.

The endpoint map commutes with translation:
\begin{equation}\label{eq:Yeh-h}
 X(Tx)=x(t)+x_0(t)=h(Xx),
 \qquad h(\xi)=\xi+x_0(t).
\end{equation}
Moreover,
\[
 P_X=N(0,t),
 \qquad
 P_{X\circ T}=N(x_0(t),t),
\]
so the Radon--Nikodym derivative is
\begin{equation}\label{eq:Yeh-ratio}
 \frac{dP_{X\circ T}}{dP_X}(\xi)
 =\exp\left\{-\frac{x_0(t)^2}{2t}\right\}
  \exp\left\{\frac{\xi x_0(t)}t\right\}.
\end{equation}

\section{Theorem 2: translation of conditional Wiener integrals}
\sourceanchor{Theorem 2, equations (3.5)--(3.6).}
\begin{theorem}[Yeh, Theorem 2]\label{thm:Yeh-2}
Let $Y$ be a real-valued Wiener-integrable functional and let $x_0$ be absolutely continuous with $x_0'\in L^2[0,t]$.  For arbitrary versions,
\begin{align}
 E^w(Y\mid X)(\xi)
 &=E^w\bigl(Y[\,\cdot+x_0]J\mid X\bigr)(\xi-x_0(t))\notag\\
 &\qquad\times
 \exp\left\{-\frac{x_0(t)^2}{2t}\right\}
 \exp\left\{\frac{\xi x_0(t)}t\right\}
 \label{eq:Yeh-main}
\end{align}
for Lebesgue-almost every $\xi$, with $J$ given by \eqref{eq:Yeh-J}.
\end{theorem}

\begin{proof}
Equation \eqref{eq:Yeh-CM} is condition $1^\circ$ of Theorem~\ref{thm:Yeh-1}; equation \eqref{eq:Yeh-h} is condition $2^\circ$; and \eqref{eq:Yeh-ratio} verifies condition $3^\circ$.  Therefore \eqref{eq:Yeh-T1-pointwise} gives exactly \eqref{eq:Yeh-main}.  Since the centered Gaussian density in \eqref{eq:Yeh-endpoint-law} is strictly positive, $P_X$ and Lebesgue measure have the same null sets.
\end{proof}

\section{The conditional Cameron--Martin exponential}
\sourceanchor{Corollary, equations (3.7)--(3.10).}
Set $Y\equiv1$.  One version of $E^w(1\mid X)$ is the constant function $1$.  Since $Y[x+x_0]=1$, Theorem~\ref{thm:Yeh-2} first gives an identity at the shifted endpoint $\xi-x_0(t)$.  Yeh then uses the fact that a translate of a Lebesgue-null set is null, replaces the endpoint variable, and obtains the following version:
\begin{align}
 &E^w\left[
 \exp\left\{-\int_0^tx_0'(s)\,dx(s)\right\}
 \middle|X\right](\xi)\notag\\
 &\quad=
 \exp\left\{\frac12\int_0^t|x_0'(s)|^2ds\right\}
 \exp\left\{-\frac{x_0(t)^2}{2t}\right\}
 \exp\left\{-\frac{\xi x_0(t)}t\right\}.
 \label{eq:Yeh-corollary}
\end{align}
The variable translation in this proof is not cosmetic: conditional expectations are equivalence classes, and a pointwise substitution must carry the exceptional set along with it.

\section{The exponential Paley--Wiener example}
\sourceanchor{Example, equations (3.11)--(3.12), and the final displayed endpoint integral.}
Let $p$ be continuous and of bounded variation and let $\lambda\in\R$.  Define
\begin{equation}\label{eq:Yeh-example-Y}
 Y[x]
 :=\exp\left\{\lambda\int_0^tp(s)\,dx(s)\right\}.
\end{equation}
Choose
\begin{equation}\label{eq:Yeh-example-x0}
 x_0(s):=-\lambda\int_0^sp(r)dr,
 \qquad
 x_0'(s)=-\lambda p(s).
\end{equation}
Then $Y[x]=\exp\{-\int_0^tx_0'(s)dx(s)\}$, so \eqref{eq:Yeh-corollary} gives
\begin{align}
 E^w(Y\mid X)(\xi)
 &=\exp\left\{\frac{\lambda^2}{2}\int_0^tp(s)^2ds\right\}
 \exp\left\{-\frac{\lambda^2}{2t}
 \left(\int_0^tp(s)ds\right)^2\right\}\notag\\
 &\qquad\times
 \exp\left\{\frac{\lambda\xi}{t}\int_0^tp(s)ds\right\}.
 \label{eq:Yeh-example-conditional}
\end{align}
Let
\[
 a:=\int_0^tp(s)ds.
\]
For every Borel set $B\subset\R$, the definition of conditional expectation and \eqref{eq:Yeh-endpoint-law} yield
\begin{align}
 &\int_{\{x:x(t)\in B\}}
 \exp\left\{\lambda\int_0^tp(s)dx(s)\right\}m_w(dx)\notag\\
 &\quad=\int_BE^w(Y\mid X)(\xi)P_X(d\xi)\notag\\
 &\quad=\exp\left\{\frac{\lambda^2}{2}\int_0^tp(s)^2ds\right\}
 \frac1{\sqrt{2\pi t}}
 \int_B\exp\left\{-\frac{(\xi-\lambda a)^2}{2t}\right\}d\xi.
 \label{eq:Yeh-final-integral}
\end{align}
The last line is obtained by combining the three endpoint-dependent exponentials:
\[
 -\frac{\xi^2}{2t}
 -\frac{\lambda^2a^2}{2t}
 +\frac{\lambda\xi a}{t}
 =-\frac{(\xi-\lambda a)^2}{2t}.
\]
Thus the weighted endpoint law is Gaussian with mean $\lambda a$ and variance $t$, multiplied by the unconditional exponential factor $\exp\{\frac12\lambda^2\int p^2\}$.

\section*{Source fidelity notes}
\begin{itemize}[leftmargin=2em]
\item The paper is bibliographically dated 1978, while the received date and some archive filenames use 1977.  The journal volume and citation determine the date used here.
\item Yeh's Cameron--Martin citation is the 1944 translation paper, not the different 1945 general-linear-transformation paper.  The exact translation reference is therefore added to the bibliography.
\item The source distinguishes the bounded-variation Riemann--Stieltjes integral from the general $L^2$ Paley--Wiener--Zygmund integral.  The reconstruction retains that distinction.
\item The equality in Theorem 1 is first an equality of integrals in the restricted sense.  The pointwise formula requires the separate absolute-continuity hypothesis $P_{X\circ T}\ll P_X$.
\item The corollary's shift of exceptional sets is part of the proof and is not suppressed by choosing a Brownian-bridge version in advance.
\item Modern Gaussian-regression or Brownian-bridge arguments give \eqref{eq:Yeh-example-conditional} immediately, but they are not substituted for Yeh's change-of-variables proof.
\end{itemize}
