\section{Target distribution and affine reference flow}
\sourceanchor{Source Sections 1--3, especially equations (1)--(11).}

Let $p^{\mathrm{base}}$ be the terminal law of a pretrained dynamical generative model and let $r:\R^d\to\R$ be a reward.  The desired fine-tuned terminal law is the exponential tilt
\begin{equation*}
 p^*(x)=\frac{p^{\mathrm{base}}(x)e^{r(x)}}{Z_r},
 \qquad
 Z_r:=\int_{\R^d}p^{\mathrm{base}}(y)e^{r(y)}\,dy.
 \tag{1}
\end{equation*}

Let $\bar X_0\sim\mathcal N(0,I)$ and $\bar X_1\sim p^{\mathrm{data}}$ be independent.  The affine reference flow is
\begin{equation*}
 \bar X_t=\beta_t\bar X_0+\alpha_t\bar X_1,
 \qquad
 \alpha_0=\beta_1=0,
 \quad
 \alpha_1=\beta_0=1.
 \tag{2}
\end{equation*}
Write $p_t$ for the law of $\bar X_t$ and
\[
 s(x,t):=\nabla_x\log p_t(x)
\]
for its score.

\subsection{Flow Matching with an arbitrary diffusion coefficient}
A Flow-Matching model uses the probability-flow ODE
\begin{equation*}
 dX_t=v(X_t,t)\,dt,
 \qquad X_0\sim\mathcal N(0,I).
 \tag{3}
\end{equation*}
The conditional velocity of the affine interpolation is
\[
 v(x,t)=\E[\dot\alpha_t\bar X_1+\dot\beta_t\bar X_0\mid \bar X_t=x].
\]
Since $\bar X_1=(x-\beta_t\bar X_0)/\alpha_t$,
\[
 v(x,t)
 =\frac{\dot\alpha_t}{\alpha_t}x
 +\left(\dot\beta_t-\frac{\dot\alpha_t}{\alpha_t}\beta_t\right)
 \E[\bar X_0\mid \bar X_t=x].
\]
The conditional-density calculation in Appendix B.4 gives
\begin{equation*}
 \E[\bar X_0\mid \bar X_t=x]=-\beta_t s(x,t),
 \tag{106}
\end{equation*}
so
\begin{equation*}
 v(x,t)=\frac{\dot\alpha_t}{\alpha_t}x
 +\beta_t\left(\frac{\dot\alpha_t}{\alpha_t}\beta_t-\dot\beta_t\right)s(x,t).
 \tag{8/107}
\end{equation*}
Define
\begin{equation*}
 \kappa_t:=\frac{\dot\alpha_t}{\alpha_t},
 \qquad
 \eta_t:=\beta_t\left(\frac{\dot\alpha_t}{\alpha_t}\beta_t-\dot\beta_t\right).
 \tag{11a}
\end{equation*}
Then the ODE drift is $\kappa_tx+\eta_ts(x,t)$.  Adding the standard Fokker--Planck correction gives, for an arbitrary state-independent diffusion coefficient $\sigma(t)$,
\begin{equation*}
 dX_t=
 \left[
 v(X_t,t)
 +\frac{\sigma(t)^2}{2\eta_t}
 \left(v(X_t,t)-\frac{\dot\alpha_t}{\alpha_t}X_t\right)
 \right]dt+\sigma(t)dB_t.
 \tag{4}
\end{equation*}
Substituting (8/107) yields the score form
\begin{equation*}
 dX_t=b(X_t,t)dt+\sigma(t)dB_t,
 \qquad
 b(x,t)=\kappa_tx+\left(\frac{\sigma(t)^2}{2}+\eta_t\right)s(x,t).
 \tag{10--11}
\end{equation*}
The cancellation in the Fokker--Planck equation shows that all choices of $\sigma$ in this family share the same one-time marginals $p_t$.

\subsection{Continuous-time DDIM and its score form}
\sourceanchor{Source Appendix B.1, equations (44)--(49), and Appendix B.3, equations (98)--(102).}

The discrete DDIM update is
\begin{equation*}
 x_{k+1}
 =\sqrt{\bar\alpha_{k+1}}
 \frac{x_k-\sqrt{1-\bar\alpha_k}\,\epsilon(x_k,k)}{\sqrt{\bar\alpha_k}}
 +\sqrt{1-\bar\alpha_{k+1}-\sigma_k^2}\,\epsilon(x_k,k)
 +\sigma_k\varepsilon_k.
 \tag{44}
\end{equation*}
For $\Delta\bar\alpha_k=\bar\alpha_{k+1}-\bar\alpha_k$, the source uses the first-order expansions
\begin{equation*}
 \sqrt{\frac{\bar\alpha_{k+1}}{\bar\alpha_k}}
 \simeq 1+\frac{\Delta\bar\alpha_k}{2\bar\alpha_k},
 \tag{45}
\end{equation*}
and
\begin{equation*}
 -\sqrt{\frac{\bar\alpha_{k+1}}{\bar\alpha_k}(1-\bar\alpha_k)}
 +\sqrt{1-\bar\alpha_{k+1}-\sigma_k^2}
 \simeq
 -\left(\frac{\Delta\bar\alpha_k}{2\bar\alpha_k}+\frac{\sigma_k^2}{2}\right)
 \frac{1}{\sqrt{1-\bar\alpha_{k+1}}}.
 \tag{46}
\end{equation*}
After introducing a step size $h$, identifying $\bar\alpha_k=\bar\alpha(kh)$ and $\sigma_k=\sqrt h\,\sigma(kh)$, and reading the result as Euler--Maruyama, the continuous sampler is
\begin{equation*}
 dX_t=
 \left[
 \frac{\dot{\bar\alpha}_t}{2\bar\alpha_t}X_t
 -\left(
 \frac{\dot{\bar\alpha}_t}{2\bar\alpha_t}
 +\frac{\sigma(t)^2}{2}
 \right)
 \frac{\epsilon(X_t,t)}{\sqrt{1-\bar\alpha_t}}
 \right]dt+\sigma(t)dB_t.
 \tag{49}
\end{equation*}
For DDIM, the affine reference coefficients are $\alpha_t=\sqrt{\bar\alpha_t}$ and $\beta_t=\sqrt{1-\bar\alpha_t}$.  The conditional-noise identity in Appendix B.3 is
\begin{equation*}
 s(x,t)=-\frac{\epsilon(x,t)}{\sqrt{1-\bar\alpha_t}}.
 \tag{9/102}
\end{equation*}
Indeed, the noised variable has the same law as
\[
 \widehat X_t=\sqrt{\bar\alpha_{1-t}}\,\bar X_0+
 \sqrt{1-\bar\alpha_{1-t}}\,\varepsilon,
\]
so the conditional predictor of $\varepsilon$ and the conditional score are the same Gaussian regression quantity up to the factor $-1/\sqrt{1-\bar\alpha_t}$.  Substituting (9/102) into (49) gives (10--11) with
\[
 \kappa_t=\eta_t=\frac{\dot{\bar\alpha}_t}{2\bar\alpha_t}.
\]

\section{Forward--backward representation and schedule invariance}
\sourceanchor{Source Appendix B.2, equations (50)--(97), Lemmas 1--2 and Propositions 3--4.}

The memorylessness proof is carried out after reversing the data-to-noise diffusion.  The paper assumes
\begin{equation*}
 \eta_t\ge 0,
 \qquad
 \int_0^1\kappa_{1-s}\,ds=+\infty,
 \qquad
 2\int_0^1\eta_{1-t'}
 \exp\left(-2\int_{t'}^1\kappa_{1-s}\,ds\right)dt'=1.
 \tag{50}
\end{equation*}
For DDIM these identities reduce to
\begin{equation*}
 \int_0^t\kappa_{1-s}\,ds=-\frac12\log\bar\alpha_{1-t},
 \qquad
 2\int_0^t\eta_{t'}e^{-2\int_{t'}^t\kappa_sds}\,dt'=1-\bar\alpha_{1-t},
 \tag{51}
\end{equation*}
and for Flow Matching to
\begin{equation*}
 \int_0^t\kappa_{1-s}\,ds=-\log\alpha_{1-t},
 \qquad
 2\int_0^t\eta_{t'}e^{-2\int_{t'}^t\kappa_sds}\,dt'=\beta_{1-t}^2.
 \tag{52--53}
\end{equation*}
The source proves the second Flow-Matching identity by integration by parts, using
\[
 \eta_{1-t'}=\beta_{1-t'}
 \left(\frac{\dot\alpha_{1-t'}}{\alpha_{1-t'}}\beta_{1-t'}-\dot\beta_{1-t'}\right)
\]
and the endpoint values $\alpha_1=1$, $\beta_1=0$.

\subsection{The canonical forward and backward pair}
Consider
\begin{align*}
 d\widetilde X_t&=-\kappa_{1-t}\widetilde X_tdt+\sqrt{2\eta_{1-t}}\,dB_t,
 &\widetilde X_0&\sim p^{\mathrm{data}},
 \tag{54}\\
 dX_t&=\bigl(\kappa_tX_t+2\eta_ts(X_t,t)\bigr)dt+\sqrt{2\eta_t}\,dB_t,
 &X_0&\sim\mathcal N(0,I).
 \tag{55}
\end{align*}
Let $\widetilde p_t$ and $p_t$ be their densities and define
$s(x,t)=\nabla\log\widetilde p_{1-t}(x)$.  Their Fokker--Planck equations are
\begin{align*}
 \partial_t\widetilde p_t
 &=\nabla\cdot(\kappa_{1-t}x\widetilde p_t)+\eta_{1-t}\Delta\widetilde p_t,
 \tag{56}\\
 \partial_tp_t
 &=\nabla\cdot\left((-\kappa_tx-2\eta_t\nabla\log\widetilde p_{1-t})p_t\right)
 +\eta_t\Delta p_t.
 \tag{57}
\end{align*}

\begin{lemma}[Source Lemma 2: variation of constants]
For
\begin{equation*}
 d\widetilde X_t=\bigl(-\kappa_{1-t}\widetilde X_t+\xi_t\bigr)dt
 +\sqrt{2\eta_{1-t}}\,dB_t,
 \tag{58}
\end{equation*}
the solution is
\begin{align*}
 \widetilde X_t={}&\widetilde X_0e^{-\int_0^t\kappa_{1-s}ds}
 +\int_0^te^{-\int_{t'}^t\kappa_{1-s}ds}\xi_{t'}dt'\\
 &+\int_0^t\sqrt{2\eta_{1-t'}}e^{-\int_{t'}^t\kappa_{1-s}ds}\,dB_{t'}.
 \tag{59}
\end{align*}
The last stochastic integral has the same law as
\[
 \left(
 2\int_0^t\eta_{1-t'}e^{-2\int_{t'}^t\kappa_{1-s}ds}dt'
 \right)^{1/2}\varepsilon,
 \qquad \varepsilon\sim\mathcal N(0,I).
 \tag{60}
\]
\end{lemma}
\begin{proof}
Multiply (58) by the integrating factor
$\exp(\int_0^t\kappa_{1-s}ds)$, apply the product rule, and integrate.  The Gaussian law in (60) follows from It\^o isometry.
\end{proof}

With $\xi=0$, condition (50) gives
\begin{equation*}
 \widetilde X_1\stackrel{d}{=}\varepsilon\sim\mathcal N(0,I).
 \tag{61}
\end{equation*}
Moreover, direct differentiation shows that $\widetilde p_{1-t}$ solves (57).  Uniqueness of the Fokker--Planck solution therefore gives the source's Proposition 3:
\begin{equation*}
 p_t=\widetilde p_{1-t}
 \qquad(0\le t\le 1).
 \tag{P3}
\end{equation*}

\subsection{Arbitrary noise schedule}
For an arbitrary $\sigma$, the paper considers
\begin{align*}
 d\widetilde X_t
 &=\left[-\kappa_{1-t}\widetilde X_t
 +\left(\frac{\sigma(1-t)^2}{2}-\eta_{1-t}\right)
 s(\widetilde X_t,1-t)\right]dt
 +\sigma(1-t)dB_t,
 \tag{63}\\
 dX_t
 &=\left[\kappa_tX_t+
 \left(\frac{\sigma(t)^2}{2}+\eta_t\right)s(X_t,t)\right]dt
 +\sigma(t)dB_t.
 \tag{64}
\end{align*}
Their Fokker--Planck equations reduce, after cancellation of score terms, to the same equations as (56)--(57).  Consequently the one-time marginals remain $\widetilde p_t$ and $p_t$ for every $\sigma$.

\begin{proposition}[Source Proposition 4: equality of path laws after time reversal]
For any sequence $0\le t_0<\cdots<t_m\le1$,
\[
 (\widetilde X_{t_0},\ldots,\widetilde X_{t_m})
 \stackrel d=
 (X_{1-t_0},\ldots,X_{1-t_m}).
\]
Equivalently, the path laws of (63) and (64) agree after reversing time.
\end{proposition}

\begin{proof}[Source proof, including its discrete approximation]
Set
\begin{align*}
 \widetilde b(x,t)&=-\kappa_{1-t}x+
 \left(\frac{\sigma(1-t)^2}{2}-\eta_{1-t}\right)s(x,1-t),
 \tag{78}\\
 b(x,t)&=\kappa_tx+
 \left(\frac{\sigma(t)^2}{2}+\eta_t\right)s(x,t),
 \tag{79}\\
 \widetilde\sigma(t)&=\sigma(1-t).
 \tag{80}
\end{align*}
Then
\[
 b(x,t)=-\widetilde b(x,1-t)+\sigma(t)^2\nabla\log p_t(x).
\]
Discretize the forward process with step $h=1/K$:
\begin{equation*}
 x_{k+1}=x_k+h\widetilde b(x_k,kh)+\sqrt h\,\widetilde\sigma(kh)\varepsilon_k.
 \tag{81}
\end{equation*}
Its path density is
\begin{equation*}
 \widetilde p(x_{0:K})
 =\widetilde p_0(x_0)\prod_{k=0}^{K-1}
 \widetilde p_{k+1\mid k}(x_{k+1}\mid x_k),
 \tag{82}
\end{equation*}
with Gaussian transition kernels.  Insert the telescoping marginal ratio:
\begin{equation*}
 \widetilde p(x_{0:K})
 =\widetilde p_K(x_K)
 \prod_{k=0}^{K-1}
 \widetilde p_{k+1\mid k}(x_{k+1}\mid x_k)
 \frac{\widetilde p_k(x_k)}{\widetilde p_{k+1}(x_{k+1})}.
 \tag{83}
\end{equation*}
The source now expands the logarithm of the marginal density by a discrete It\^o formula:
\begin{align*}
 \log\widetilde p(x_{k+1},(k+1)h)
 \simeq{}&\log\widetilde p(x_k,kh)
 +h\left(\partial_t\log\widetilde p
 +\frac{\widetilde\sigma(kh)^2}{2}\Delta\log\widetilde p\right)(x_k,kh)\\
 &+\langle\nabla\log\widetilde p(x_k,kh),x_{k+1}-x_k\rangle
 +O(h^{3/2}).
 \tag{84--85}
\end{align*}
Using the Fokker--Planck equation
\begin{equation*}
 \partial_t\widetilde p_t
 =\nabla\cdot\left[
 \left(-\widetilde b+\frac{\widetilde\sigma^2}{2}\nabla\log\widetilde p_t\right)
 \widetilde p_t
 \right],
 \tag{87}
\end{equation*}
together with Taylor expansions of $\nabla\log\widetilde p$ and $\widetilde b$, the source obtains the following expression.  To display its Gaussian residual without obscuring the identity, set
\begin{equation*}
 \mathcal R_{k+1}:=
 x_{k+1}-x_k-h\widetilde b(x_{k+1},(k+1)h)
 +h\widetilde\sigma((k+1)h)^2
 \nabla\log\widetilde p(x_{k+1},(k+1)h).
\end{equation*}
Then
\begin{align*}
 &\log\widetilde p_{k+1\mid k}(x_{k+1}\mid x_k)
 -\left[
 \log\widetilde p(x_{k+1},(k+1)h)-\log\widetilde p(x_k,kh)
 \right]\\
 &\qquad=
 -\frac d2\log(2\pi h\widetilde\sigma((k+1)h)^2)
 -\frac{\|\mathcal R_{k+1}\|^2}
 {2h\widetilde\sigma((k+1)h)^2}
 +O(h^{3/2}).
 \tag{93}
\end{align*}
By Bayes' rule, the right side is the log-density, up to $O(h^{3/2})$, of the reverse update
\begin{equation*}
 x_k=x_{k+1}+h\left[
 -\widetilde b(x_{k+1},(k+1)h)
 +\widetilde\sigma((k+1)h)^2\nabla\log\widetilde p(x_{k+1},(k+1)h)
 \right]
 +\sqrt h\,\widetilde\sigma((k+1)h)\varepsilon_{k+1}.
 \tag{95}
\end{equation*}
After $y_k=x_{K-k}$ and the identity relating $b$ and $\widetilde b$, this becomes
\begin{equation*}
 y_{k+1}=y_k+hb(y_k,kh)+\sqrt h\,\sigma(kh)\varepsilon_k,
 \tag{96}
\end{equation*}
which is the Euler--Maruyama scheme for (64).  Substitution into (83) gives the reverse factorization
\begin{equation*}
 \widetilde p(x_{0:K})\simeq
 \widetilde p_K(x_K)
 \prod_{k=0}^{K-1}\widetilde p_{k+1\mid k}(x_{k+1}\mid x_k),
 \tag{97}
\end{equation*}
and the source concludes equality of the continuous trajectory laws after time reversal.
\end{proof}

\begin{remark}[Logical status of the source proof of Proposition 4]
The paper itself says that this derivation is ``still in discrete time, and hence not completely formal.''  The calculation identifies the reverse Euler--Maruyama transition only up to local $O(h^{3/2})$ errors.  Passing from these local approximations to exact equality of continuous path laws requires a convergence or time-reversal theorem not supplied in the paper.  The remainder of the paper uses Proposition 4 as exact.
\end{remark}

\section{Control-affine stochastic optimal control and the initial-value bias}
\sourceanchor{Source Sections 4.1--4.2, equations (12)--(23), and Appendix C.4, equations (132)--(138).}

For running cost $f$, terminal cost $g$, base drift $b$, and diffusion matrix $\sigma(t)$, the control problem is
\begin{align*}
 \inf_{u\in\mathcal U}\quad
 &\E\left[
 \int_0^1\left(\frac12\|u(X_t^u,t)\|^2+f(X_t^u,t)\right)dt
 +g(X_1^u)
 \right],
 \tag{12}\\
 dX_t^u&=\bigl(b(X_t^u,t)+\sigma(t)u(X_t^u,t)\bigr)dt+\sigma(t)dB_t,
 \qquad X_0^u\sim p_0.
 \tag{13}
\end{align*}
Define the cost-to-go and value function by
\begin{align*}
 J(u;x,t)
 &:=\E\left[
 \int_t^1\left(\frac12\|u(X_s,s)\|^2+f(X_s,s)\right)ds+g(X_1)
 \mid X_t=x
 \right],
 \tag{14}\\
 V(x,t)&:=\inf_uJ(u;x,t)=J(u^*;x,t).
 \tag{15}
\end{align*}
The linearly solvable SOC identity is
\begin{equation*}
 V(x,t)=-\log\E_{P^{\mathrm{base}}}
 \left[
 \exp\left(-\int_t^1f(X_s,s)ds-g(X_1)\right)
 \middle|X_t=x
 \right],
 \tag{16}
\end{equation*}
and the optimal feedback is
\begin{equation*}
 u^*(x,t)=-\sigma(t)^\top\nabla_xV(x,t)
 =-\sigma(t)^\top\nabla_xJ(u^*;x,t).
 \tag{17}
\end{equation*}

\subsection{The control energy is conditional path-space KL}
The source invokes the following Girsanov statement.  If
\begin{align*}
 dX_t&=b_1(X_t,t)dt+\sigma(X_t,t)dB_t,\\
 dY_t&=(b_1(Y_t,t)+b_2(Y_t,t))dt+\sigma(Y_t,t)dB_t
\end{align*}
start from the same deterministic initial state and admit unique strong solutions, then
\begin{equation*}
 \frac{dP^X}{dP^Y}(Y)
 =\exp\left(
 -\int_0^T\sigma^{-1}b_2(Y_t,t)\,dB_t
 -\frac12\int_0^T\|\sigma^{-1}b_2(Y_t,t)\|^2dt
 \right).
 \tag{134}
\end{equation*}
For $b_2=\sigma u$, conditioning on $X_0=x_0$ gives
\begin{equation*}
 \log\frac{dP^{\mathrm{base}}_{\mid x_0}}{dP^u_{\mid x_0}}(X^u)
 =-\int_0^1u(X_t^u,t)\,dB_t
 -\frac12\int_0^1\|u(X_t^u,t)\|^2dt.
 \tag{137}
\end{equation*}
Taking expectation under $P^u_{\mid x_0}$ and using the martingale property of the stochastic integral yields
\begin{equation*}
 \KL(P^u(\cdot\mid X_0)\|P^{\mathrm{base}}(\cdot\mid X_0))
 =\E_{P^u}\left[
 \frac12\int_0^1\|u(X_t^u,t)\|^2dt
 \middle|X_0
 \right].
 \tag{18/138}
\end{equation*}
Thus the SOC objective is the conditional path-space Gibbs variational problem
\begin{equation*}
 \sup_u\E_{X_0\sim p_0}\left[
 \E_{P^u(\cdot\mid X_0)}
 \left(-\int_0^1f(X_t,t)dt-g(X_1)\right)
 -\KL(P^u(\cdot\mid X_0)\|P^{\mathrm{base}}(\cdot\mid X_0))
 \right].
 \tag{19}
\end{equation*}

\subsection{Conditional Gibbs tilt and the bias term}
Conditioned on $X_0$, the optimizer has density
\begin{equation*}
 P^*(dX\mid X_0)
 \propto
 P^{\mathrm{base}}(dX\mid X_0)
 \exp\left(-\int_0^1f(X_t,t)dt-g(X_1)\right).
 \tag{20}
\end{equation*}
Its conditional normalizer is
\begin{equation*}
 \E_{P^{\mathrm{base}}(\cdot\mid X_0)}
 \left[
 \exp\left(-\int_0^1f(X_t,t)dt-g(X_1)\right)
 \right]
 =e^{-V(X_0,0)}.
 \tag{21}
\end{equation*}
Multiplying by the fixed initial law therefore gives
\begin{equation*}
 P^*(dX)=P^{\mathrm{base}}(dX)
 \exp\left(-\int_0^1f(X_t,t)dt-g(X_1)+V(X_0,0)\right).
 \tag{22}
\end{equation*}
For reward fine-tuning, $f=0$ and $g=-r$, so the endpoint pair satisfies
\begin{equation*}
 p^*(x_0,x_1)=p^{\mathrm{base}}(x_0,x_1)
 e^{r(x_1)+V(x_0,0)}.
 \tag{23}
\end{equation*}
Unless the base endpoints are independent, integrating out $x_0$ does not in general produce (1).  This is the initial-value-function bias.

\section{The memoryless schedule}
\sourceanchor{Source Definition 1, equation (24), Proposition 1, equation (25), and Appendix D.1, equations (139)--(154).}

\begin{definition}[Source Definition 1]
A process in the family (10--11) is \emph{memoryless} when $X_0$ and $X_1$ are independent.
\end{definition}
If the base process is memoryless, then (23) factors as
\begin{align*}
 p^*(x_1)
 &=\int p^{\mathrm{base}}(x_0)p^{\mathrm{base}}(x_1)
 e^{r(x_1)+V(x_0,0)}dx_0\\
 &\propto p^{\mathrm{base}}(x_1)e^{r(x_1)},
 \tag{24}
\end{align*}
so the terminal law is the desired tilt (1).

\begin{proposition}[Source Proposition 1: memoryless noise schedules]
Within the sampler family (10--11), the source characterizes memorylessness by
\begin{equation*}
 \sigma(t)^2=2\eta_t+\chi(t),
 \tag{25a}
\end{equation*}
where, for every fixed $t>0$,
\begin{equation*}
 \lim_{t'\downarrow0}
 \alpha_{t'}
 \exp\left(
 -\int_{t'}^t\frac{\chi(s)}{2\beta_s^2}ds
 \right)=0.
 \tag{25b}
\end{equation*}
The canonical choice is $\chi=0$, hence
\begin{equation*}
 \sigma(t)=\sqrt{2\eta_t}.
 \tag{25c}
\end{equation*}
\end{proposition}

\begin{proof}[Reconstruction of Appendix D.1]
The proof uses the time-reversed forward process (63).  Since the affine marginal $p_t$ is the law of $\beta_t\bar X_0+\alpha_t\bar X_1$, its density is
\begin{equation*}
 p_t(x)=\int_{\R^d}
 \frac{\exp(-\|x-\alpha_ty\|^2/(2\beta_t^2))}{(2\pi\beta_t^2)^{d/2}}
 p^{\mathrm{data}}(y)dy.
 \tag{139}
\end{equation*}
Differentiation gives
\begin{equation*}
 s(x,t)=-\frac{x-\alpha_t\xi_t(x)}{\beta_t^2},
 \tag{140}
\end{equation*}
where
\begin{equation*}
 \xi_t(x):=
 \frac{\int y\exp(-\|x-\alpha_ty\|^2/(2\beta_t^2))p^{\mathrm{data}}(y)dy}
 {\int \exp(-\|x-\alpha_ty\|^2/(2\beta_t^2))p^{\mathrm{data}}(y)dy}.
 \tag{141}
\end{equation*}
Substituting (140) into the forward equation gives
\begin{equation*}
 d\widetilde X_t=
 \left[
 -\kappa_{1-t}\widetilde X_t
 -\frac{\sigma(1-t)^2-2\eta_{1-t}}{2\beta_{1-t}^2}
 \bigl(\widetilde X_t-\alpha_{1-t}\xi_{1-t}(\widetilde X_t)\bigr)
 \right]dt
 +\sigma(1-t)dB_t.
 \tag{142}
\end{equation*}
Apply the variation-of-constants formula with the modified linear coefficient.  After the algebraic identity
\begin{equation*}
 \kappa_{1-s}
 +\frac{\sigma(1-s)^2-2\eta_{1-s}}{2\beta_{1-s}^2}
 =\frac{\sigma(1-s)^2}{2\beta_{1-s}^2}
 +\frac{\dot\beta_{1-s}}{\beta_{1-s}},
 \tag{144}
\end{equation*}
one obtains the integrating factor
\begin{equation*}
 \exp\left(
 -\int_{t'}^t
 \left[
 \kappa_{1-s}+\frac{\sigma(1-s)^2-2\eta_{1-s}}{2\beta_{1-s}^2}
 \right]ds
 \right)
 =
 \exp\left(-\int_{t'}^t\frac{\sigma(1-s)^2}{2\beta_{1-s}^2}ds\right)
 \frac{\beta_{1-t}}{\beta_{1-t'}}.
 \tag{146}
\end{equation*}
Now write $\sigma(1-s)^2=2\eta_{1-s}+\chi(1-s)$.  The source simplifies the coefficient further to
\begin{equation*}
 \exp\left(-\int_{t'}^t\frac{\chi(1-s)}{2\beta_{1-s}^2}ds\right)
 \frac{\alpha_{1-t}}{\alpha_{1-t'}}.
 \tag{148}
\end{equation*}
Substitution into the explicit solution gives
\begin{align*}
 \widetilde X_t={}&
 \widetilde X_0
 \exp\left(-\int_0^t\frac{\chi(1-s)}{2\beta_{1-s}^2}ds\right)
 \frac{\alpha_{1-t}}{\alpha_1}\\
 &+\alpha_{1-t}\int_0^t
 \exp\left(-\int_{t'}^t\frac{\chi(1-s)}{2\beta_{1-s}^2}ds\right)
 \frac{\chi(1-t')}{2\beta_{1-t'}^2}
 \xi_{1-t'}(\widetilde X_{t'})dt'\\
 &+\int_0^t
 \left[
 2\beta_{1-t'}
 \left(\frac{\dot\alpha_{1-t'}}{\alpha_{1-t'}}\beta_{1-t'}-\dot\beta_{1-t'}\right)
 +\chi(1-t')
 \right]^{1/2}\\
 &\hspace{2.4cm}\times
 \exp\left(-\int_{t'}^t\frac{\chi(1-s)}{2\beta_{1-s}^2}ds\right)
 \frac{\alpha_{1-t}}{\alpha_{1-t'}}dB_{t'}.
 \tag{150}
\end{align*}
Taking $t\uparrow1$, condition (25b), after reversing time, becomes
\begin{equation*}
 \lim_{t\uparrow1}
 \alpha_{1-t}
 \exp\left(-\int_{t'}^t\frac{\chi(1-s)}{2\beta_{1-s}^2}ds\right)=0
 \qquad(0\le t'<1).
 \tag{152}
\end{equation*}
This annihilates both the coefficient of $\widetilde X_0$ and the deterministic integral containing $\xi$.  The endpoint therefore reduces to the stochastic integral
\begin{align*}
 \widetilde X_1
 =\lim_{t\uparrow1}\int_0^t
 &\left[
 2\beta_{1-t'}
 \left(\frac{\dot\alpha_{1-t'}}{\alpha_{1-t'}}\beta_{1-t'}-\dot\beta_{1-t'}\right)
 +\chi(1-t')
 \right]^{1/2}\\
 &\times
 \exp\left(-\int_{t'}^t\frac{\chi(1-s)}{2\beta_{1-s}^2}ds\right)
 \frac{\alpha_{1-t}}{\alpha_{1-t'}}dB_{t'}.
 \tag{154}
\end{align*}
The source concludes that $\widetilde X_1$ is independent of $\widetilde X_0$.  Proposition 4 identifies
$(\widetilde X_0,\widetilde X_1)$ in law with $(X_1,X_0)$, so $X_0$ and $X_1$ are independent.

For necessity, the source returns to (150): if the limit in (152) is nonzero for some $t'$, the first terms retain dependence on $\widetilde X_0$, and it concludes that memorylessness generally fails.
\end{proof}

\begin{remark}[Qualification of the ``only if'' direction]
The sufficiency calculation explicitly removes the initial variable.  The source's necessity paragraph is weaker: it says that a nonzero coefficient means the endpoint ``in general'' is not independent.  A nonzero measurable dependence does not by itself rule out independence for every exceptional data distribution, and the proof does not quantify the model class over which necessity is asserted.  Thus the displayed characterization is the paper's stated theorem, but its necessity argument is heuristic unless additional nondegeneracy assumptions are supplied.
\end{remark}

\section{Fine-tuning with the memoryless schedule}
\sourceanchor{Source Theorem 1 and Appendix D.2, equations (155)--(179).}

The paper uses the Hamilton--Jacobi--Bellman equation in the following form.  For generator
\begin{equation*}
 \mathcal L
 =\frac12\sum_{i,j}(\sigma\sigma^\top)_{ij}(t)\partial_{x_i}\partial_{x_j}
 +\sum_i b_i(x,t)\partial_{x_i},
 \tag{155}
\end{equation*}
the value function satisfies
\begin{equation*}
 \partial_tV=-\mathcal LV+\frac12\|\sigma^\top\nabla V\|^2-f,
 \qquad V(x,1)=g(x).
 \tag{156}
\end{equation*}

\begin{theorem}[Source Theorem 1: fine-tuning recipe]
Within the family (10--11), to fine-tune by the SOC problem with $f=0$, $g=-r$ and still permit sampling afterward with an arbitrary equivalent noise schedule, the SOC problem is solved using
\[
 \sigma(t)=\sqrt{2\eta_t}.
\]
The resulting controlled score is
\begin{equation*}
 s^*(x,t)=s(x,t)+\frac{u^*(x,t)}{\sqrt{2\eta_t}},
 \tag{170}
\end{equation*}
and this score can then be placed into any sampler in the common marginal family.
\end{theorem}

\begin{proof}[Reconstruction of Appendix D.2]
Start forward noising processes from $p^{\mathrm{base}}$ and $p^*$:
\begin{align*}
 d\widetilde X_t&=\widetilde b(\widetilde X_t,t)dt+\sigma(t)dB_t,
 &\widetilde X_0&\sim p^{\mathrm{base}},
 \tag{157}\\
 d\widetilde X_t^*&=\widetilde b^*(\widetilde X_t^*,t)dt+\sigma(t)dB_t,
 &\widetilde X_0^*&\sim p^*.
 \tag{158}
\end{align*}
Let $\widetilde p_t$, $\widetilde p_t^*$ be their densities and define
$\Psi=-\log\widetilde p$, $\Psi^*=-\log\widetilde p^*$.  Inserting the score-form drifts into the Fokker--Planck equations cancels the arbitrary $\sigma$ and gives
\begin{align*}
 \partial_t\widetilde p_t
 &=\nabla\cdot(\kappa_{1-t}x\widetilde p_t)+\nabla\cdot(\eta_{1-t}\nabla\widetilde p_t),\\
 \partial_t\widetilde p_t^*
 &=\nabla\cdot(\kappa_{1-t}x\widetilde p_t^*)+\nabla\cdot(\eta_{1-t}\nabla\widetilde p_t^*).
 \tag{161}
\end{align*}
The Hopf--Cole calculation yields
\begin{align*}
 -\partial_t\Psi
 &=\kappa_{1-t}d-\kappa_{1-t}\langle x,\nabla\Psi\rangle
 +\eta_{1-t}(-\Delta\Psi+\|\nabla\Psi\|^2),
 \tag{163}\\
 -\partial_t\Psi^*
 &=\kappa_{1-t}d-\kappa_{1-t}\langle x,\nabla\Psi^*\rangle
 +\eta_{1-t}(-\Delta\Psi^*+\|\nabla\Psi^*\|^2).
 \tag{164}
\end{align*}
Set $\widehat\Psi=\Psi^*-\Psi$.  Subtracting and expanding
$\|\nabla(\Psi+\widehat\Psi)\|^2-\|\nabla\Psi\|^2$ gives
\begin{equation*}
 -\partial_t\widehat\Psi
 =\left\langle
 -\kappa_{1-t}x+2\eta_{1-t}\nabla\Psi,
 \nabla\widehat\Psi
 \right\rangle
 +\eta_{1-t}(-\Delta\widehat\Psi+\|\nabla\widehat\Psi\|^2).
 \tag{165}
\end{equation*}
At $t=0$,
\begin{equation*}
 \widehat\Psi(x,0)
 =-\log p^*(x)+\log p^{\mathrm{base}}(x)
 =-r(x)+\log Z_r.
 \tag{165b}
\end{equation*}
Define $V(x,t)=\widehat\Psi(x,1-t)$.  Time reversal changes (165) into
\begin{equation*}
 \partial_tV
 =\langle-\kappa_tx-2\eta_ts(x,t),\nabla V\rangle
 +\eta_t(-\Delta V+\|\nabla V\|^2),
 \qquad
 V(x,1)=-r(x)+\log Z_r.
 \tag{166$'$}
\end{equation*}
This is the HJB equation for
\begin{align*}
 \inf_u\quad
 &\E\left[
 \frac12\int_0^1\|u(X_t^u,t)\|^2dt-r(X_1^u)+\log Z_r
 \right],
 \tag{167}\\
 dX_t^u&=\left(\kappa_tX_t^u+2\eta_ts(X_t^u,t)+\sqrt{2\eta_t}\,u(X_t^u,t)\right)dt
 +\sqrt{2\eta_t}\,dB_t.
 \tag{168}
\end{align*}
Hence
\begin{align*}
 u^*(x,t)
 &=-\sqrt{2\eta_t}\,\nabla V(x,t)\\
 &=\sqrt{2\eta_t}\,[s^*(x,t)-s(x,t)],
 \tag{169}
\end{align*}
which is (170).

The backward sampler associated with the target score is
\begin{equation*}
 dX_t^*
 =\left[
 \kappa_tX_t^*+
 \left(\frac{\sigma(t)^2}{2}+\eta_t\right)s^*(X_t^*,t)
 \right]dt+\sigma(t)dB_t.
 \tag{171}
\end{equation*}
Substitute (170):
\begin{equation*}
 dX_t^*
 =\left[
 b(X_t^*,t)
 +\frac{\sigma(t)^2/2+\eta_t}{\sqrt{2\eta_t}}u^*(X_t^*,t)
 \right]dt+\sigma(t)dB_t.
 \tag{173}
\end{equation*}
Thus the SOC control is learned under the memoryless schedule, while the target score $s^*$ is schedule-independent and can be used in any sampler of the family.

For DDIM, using
\[
 u^*(x,t)=
 -\sqrt{\frac{\dot{\bar\alpha}_t}{\bar\alpha_t(1-\bar\alpha_t)}}
 \bigl(\epsilon^*(x,t)-\epsilon^{\mathrm{base}}(x,t)\bigr),
 \tag{26}
\]
the drift algebra reduces (173) to
\begin{equation*}
 dX_t^*
 =\left[
 \frac{\dot{\bar\alpha}_t}{2\bar\alpha_t}X_t^*
 -\left(
 \frac{\dot{\bar\alpha}_t}{2\bar\alpha_t}+\frac{\sigma(t)^2}{2}
 \right)
 \frac{\epsilon^*(X_t^*,t)}{\sqrt{1-\bar\alpha_t}}
 \right]dt+\sigma(t)dB_t,
 \tag{176}
\end{equation*}
which is the DDIM family with the target predictor $\epsilon^*$.

For Flow Matching,
\[
 u^*(x,t)=
 \sqrt{\frac{2}{\beta_t(\frac{\dot\alpha_t}{\alpha_t}\beta_t-\dot\beta_t)}}
 \bigl(v^*(x,t)-v^{\mathrm{base}}(x,t)\bigr),
 \tag{27}
\]
and (173) becomes
\begin{equation*}
 dX_t^*=
 \left[
 v^*(X_t^*,t)
 +\frac{\sigma(t)^2}{2\beta_t(\frac{\dot\alpha_t}{\alpha_t}\beta_t-\dot\beta_t)}
 \left(v^*(X_t^*,t)-\frac{\dot\alpha_t}{\alpha_t}X_t^*\right)
 \right]dt+\sigma(t)dB_t,
 \tag{179}
\end{equation*}
which is the Flow-Matching sampler (4) with $v=v^*$.
\end{proof}

\begin{remark}[Sign inconsistency in source equation (166)]
Equation (165) and the definition $p^*\propto p^{\mathrm{base}}e^r$ give
$\widehat\Psi(x,0)=-r(x)+\log Z_r$.  The objective in source equation (167) has the same terminal cost.  However, the displayed terminal condition in source equation (166) is printed with the opposite sign, $r(x)-\log Z_r$.  The reconstruction uses the sign forced by equations (165), (167), and the desired tilt.
\end{remark}

\begin{remark}[Analytic assumptions used implicitly]
The proof differentiates positive densities, applies Fokker--Planck equations pointwise, invokes uniqueness for an HJB equation, and identifies optimal feedbacks from classical solutions.  The paper does not collect the regularity, growth, invertibility, and integrability hypotheses needed for these steps into a single theorem statement.
\end{remark}

\section{Continuous adjoints}
\sourceanchor{Source Section 5.1.1, equations (28)--(32), and Appendix E.1, Proposition 6, Lemma 5, equations (180)--(192).}

For a sampled controlled path $X\sim P^u$, define the pathwise cost
\begin{equation*}
 \mathcal L(u;X)
 =\int_0^T\left(\frac12\|u(X_t,t)\|^2+f(X_t,t)\right)dt+g(X_T).
 \tag{28}
\end{equation*}
The adjoint state is the spatial derivative of the future pathwise cost,
\begin{equation*}
 a(t;X,u)
 :=\nabla_{X_t}\left[
 \int_t^T\left(\frac12\|u(X_{t'},t')\|^2+f(X_{t'},t')\right)dt'
 +g(X_T)
 \right].
 \tag{29}
\end{equation*}
It solves backward
\begin{align*}
 \frac d{dt}a(t;X,u)
 &=-\left[
 \nabla_x(b(X_t,t)+\sigma(t)u(X_t,t))^\top a(t;X,u)
 +\nabla_x\left(f(X_t,t)+\frac12\|u(X_t,t)\|^2\right)
 \right],
 \tag{30}\\
 a(T;X,u)&=\nabla g(X_T).
 \tag{31}
\end{align*}
Conditioning (29) gives
\begin{equation*}
 \E[a(t;X,u)\mid X_t=x]=\nabla_xJ(u;x,t).
 \tag{29b}
\end{equation*}

\begin{lemma}[Source Lemma 5]
Let $v$ be a fixed vector field and let $X^{u_\theta}$ solve the controlled SDE.  The solution of (30)--(31) is (29), and
\begin{equation*}
 \nabla_\theta\mathcal L(v;X^{u_\theta})
 =\int_0^T
 (\nabla_\theta u_\theta)(X_t^{u_\theta},t)^\top
 \sigma(t)^\top a(t)dt.
 \tag{185}
\end{equation*}
\end{lemma}

\begin{proof}[Source Lagrange-multiplier calculation]
Write $\widetilde b_\theta=b+\sigma u_\theta$ and subtract the SDE constraint from the cost using a multiplier $a_t$:
\begin{align*}
 \mathcal L(v;X^{u_\theta})
 ={}&\int_0^T\left(\frac12\|v(X_t,t)\|^2+f(X_t,t)\right)dt+g(X_T)\\
 &-\int_0^T\left\langle a_t,dX_t-\widetilde b_\theta(X_t,t)dt-\sigma(t)dB_t\right\rangle.
 \tag{186}
\end{align*}
Stochastic integration by parts gives
\begin{equation*}
 \int_0^T\langle a_t,dX_t\rangle
 =\langle a_T,X_T\rangle-\langle a_0,X_0\rangle
 -\int_0^T\left\langle X_t,\frac{da_t}{dt}\right\rangle dt.
 \tag{187}
\end{equation*}
Differentiate first with respect to the initial state.  Choosing $a$ so that all state-variation terms cancel gives exactly (30)--(31), and the remaining boundary term is
\begin{equation*}
 \nabla_{x_0}\mathcal L(v;X^{u_\theta})=a_0.
 \tag{190}
\end{equation*}
The same calculation for a parameter variation has $\nabla_\theta X_0=0$ and leaves only the drift derivative:
\begin{align*}
 \nabla_\theta\mathcal L(v;X^{u_\theta})
 &=\int_0^T
 (\nabla_\theta\widetilde b_\theta)(X_t^{u_\theta},t)^\top a_tdt\\
 &=\int_0^T
 (\nabla_\theta u_\theta)(X_t^{u_\theta},t)^\top\sigma(t)^\top a_tdt.
 \tag{192}
\end{align*}
\end{proof}

\begin{proposition}[Source Proposition 6: continuous-adjoint gradient]
For the same control in both the trajectory and the cost,
\begin{equation*}
 \frac{d}{d\theta}\E[\mathcal L(u_\theta;X^{u_\theta})]
 =\E\int_0^T
 \left\langle
 u_\theta(X_t,t)+\sigma(t)^\top a(t;X,u_\theta),
 \partial_\theta u_\theta(X_t,t)
 \right\rangle dt.
 \tag{32}
\end{equation*}
\end{proposition}
\begin{proof}
Split the derivative into the explicit derivative of the quadratic control cost and the derivative through the trajectory while freezing the cost vector field.  The first is
$\E\int\langle u,\partial_\theta u\rangle dt$; Lemma 5 gives the second.  Adding them gives (32).
\end{proof}

\section{Basic Adjoint Matching}
\sourceanchor{Source Proposition 2 and Appendix E.2, equations (193)--(202), Lemma 6.}

Freeze the trajectory-generating control by $\bar u=\operatorname{stopgrad}(u)$ and define
\begin{equation*}
 \mathcal L_{\mathrm{Basic-AM}}(u;X)
 :=\frac12\int_0^T
 \|u(X_t,t)+\sigma(t)^\top a(t;X,\bar u)\|^2dt,
 \qquad X\sim P^{\bar u}.
 \tag{34}
\end{equation*}

\begin{proposition}[Source Proposition 2]
The parameter gradient of the expected basic Adjoint Matching loss equals the continuous-adjoint gradient (32).  Moreover, its only functional critical point is the optimal control $u^*$.
\end{proposition}

\begin{proof}
Because $X$ and $a(t;X,\bar u)$ are frozen with respect to the differentiated parameters,
\begin{align*}
 \nabla_\theta\mathcal L_{\mathrm{Basic-AM}}
 &=\int_0^T
 \left\langle
 u(X_t,t)+\sigma(t)^\top a(t;X,\bar u),
 \partial_\theta u(X_t,t)
 \right\rangle dt,
 \tag{193--194}
\end{align*}
which is (32).

For the functional first variation, let $v$ be arbitrary.  Then
\begin{align*}
 \left.\frac d{d\varepsilon}\right|_{\varepsilon=0}
 \E\mathcal L_{\mathrm{Basic-AM}}(u+\varepsilon v;X^{\bar u})
 &=\E\int_0^T
 \left\langle
 v(X_t,t),u(X_t,t)+\sigma(t)^\top a(t;X,\bar u)
 \right\rangle dt\\
 &=\E\int_0^T
 \left\langle
 v(X_t,t),u(X_t,t)+\sigma(t)^\top
 \E[a(t;X,\bar u)\mid X_t]
 \right\rangle dt.
 \tag{195}
\end{align*}
Hence a critical point satisfies
\begin{equation*}
 u(x,t)=-\sigma(t)^\top\E[a(t;X^u,u)\mid X_t^u=x]
 =-\sigma(t)^\top\nabla_xJ(u;x,t).
 \tag{196}
\end{equation*}
It remains to identify every solution of this fixed-point relation.

By the dynamic-programming identity,
\begin{align*}
 J(u;x,t)
 ={}&\E[J(u;X_{t+\Delta t}^u,t+\Delta t)\mid X_t^u=x]\\
 &+\E\left[
 \int_t^{t+\Delta t}
 \left(\frac12\|u(X_s^u,s)\|^2+f(X_s^u,s)\right)ds
 \middle|X_t^u=x
 \right].
 \tag{197}
\end{align*}
Divide by $\Delta t$ and let $\Delta t\downarrow0$.  With controlled generator
\begin{equation*}
 \mathcal T^u\phi
 =\partial_t\phi+\langle\nabla\phi,b+\sigma u\rangle
 +\operatorname{Tr}\left(\frac{\sigma\sigma^\top}{2}\nabla^2\phi\right),
 \tag{199}
\end{equation*}
one gets
\begin{equation*}
 0=\mathcal T^uJ(u;x,t)+\frac12\|u(x,t)\|^2+f(x,t).
 \tag{200}
\end{equation*}
Using $u=-\sigma^\top\nabla J$,
\begin{equation*}
 \langle\nabla J,\sigma u\rangle+\frac12\|u\|^2
 =-\frac12\|\sigma^\top\nabla J\|^2.
 \tag{201}
\end{equation*}
Therefore
\begin{equation*}
 0=\partial_tJ+\langle\nabla J,b\rangle
 +\operatorname{Tr}\left(\frac{\sigma\sigma^\top}{2}\nabla^2J\right)
 -\frac12\|\sigma^\top\nabla J\|^2+f,
 \qquad J(x,T)=g(x).
 \tag{202}
\end{equation*}
This is the HJB equation (156).  By the uniqueness of its solution, $J=V$, so (196) gives
$u=-\sigma^\top\nabla V=u^*$.
\end{proof}

\section{Lean Adjoint Matching}
\sourceanchor{Source Section 5.2, equations (35)--(39), and Appendix E.3, equations (203)--(218), Propositions 7--8.}

At the optimum of the least-squares regression (34), the fitted control equals the conditional mean of its target:
\begin{equation*}
 u^*(x,t)=
 \E[-\sigma(t)^\top a(t;X,u^*)\mid X_t=x].
 \tag{35}
\end{equation*}
Multiplication by the control Jacobian gives
\begin{equation*}
 \E\left[
 u^*(x,t)^\top\nabla_xu^*(x,t)
 +a(t;X,u^*)^\top\sigma(t)\nabla_xu^*(x,t)
 \middle|X_t=x
 \right]=0.
 \tag{36}
\end{equation*}
These are precisely the control-dependent terms in the full adjoint equation whose conditional expectation vanishes at the optimum.  The source therefore defines the lean adjoint by removing them:
\begin{align*}
 \mathcal L_{\mathrm{AM}}(u;X)
 &:=\frac12\int_0^T
 \|u(X_t,t)+\sigma(t)^\top\widetilde a(t;X)\|^2dt,
 \qquad X\sim P^{\bar u},
 \tag{37}\\
 \frac d{dt}\widetilde a(t;X)
 &=-\left[
 \nabla_xb(X_t,t)^\top\widetilde a(t;X)+\nabla_xf(X_t,t)
 \right],
 \tag{38}\\
 \widetilde a(T;X)&=\nabla g(X_T).
 \tag{39}
\end{align*}

\begin{proposition}[Source Proposition 8: uniqueness of the conditional-adjoint equation]
Fix a control $v$.  The integral equation
\begin{align*}
 Y_t=\E\Bigg[\int_t^T
 &\left(
 \nabla_x(b(X_s^v,s)+\sigma(s)v(X_s^v,s))^\top Y_s
 +\nabla_x\left(f(X_s^v,s)+\frac12\|v(X_s^v,s)\|^2\right)
 \right)ds\\
 &+\nabla g(X_T^v)
 \ \Bigg|\ X_t^v\Bigg]
 \tag{215}
\end{align*}
has at most one solution.
\end{proposition}

\begin{proof}[Source Gronwall argument]
If $Y^1,Y^2$ are solutions, subtraction cancels the inhomogeneous terms:
\begin{equation*}
 Y_t^1-Y_t^2
 =\E\left[
 \int_t^T
 \nabla_x(b(X_s^v,s)+\sigma(s)v(X_s^v,s))^\top
 (Y_s^1-Y_s^2)ds
 \middle|X_t^v
 \right].
 \tag{216$'$}
\end{equation*}
The paper's displayed equations (216)--(218) retain only $\nabla b$ in this difference; following that display, conditional Jensen and Cauchy--Schwarz give
\begin{align*}
 \|Y_t^1-Y_t^2\|
 &\le\int_t^T
 \E[\|Y_s^1-Y_s^2\|^2\mid X_t^v]^{1/2}
 \E[\|\nabla b(X_s^v,s)\|^2\mid X_t^v]^{1/2}ds.
 \tag{217}
\end{align*}
Define the running conditional supremum
\[
 F(t):=\sup_{t'\in[0,t]}
 \E[\|Y_{t'}^1-Y_{t'}^2\|^2\mid X_{t'}^v]^{1/2}.
\]
The source rewrites (217) into an inequality of Gronwall type
\begin{equation*}
 F(t)\le\int_t^T F(s)G(s)ds,
 \tag{218}
\end{equation*}
for the corresponding conditional bound $G$ on $\nabla b$.  Gronwall's inequality yields $F=0$, hence $Y^1=Y^2$.
\end{proof}

\begin{remark}[Notational and algebraic issues in Proposition 8]
The statement uses $X^v$, while source equations (216)--(218) switch to $X^*$ without explanation.  More importantly, subtracting two solutions of (215) should retain the full Jacobian $\nabla(b+\sigma v)$, not only $\nabla b$, unless $v$ is independent of $x$.  The source proof also suppresses the precise integrability and boundedness assumptions needed for the conditional Gronwall estimate.  These are recorded here rather than silently repaired.
\end{remark}

\begin{proposition}[Source Proposition 7: critical point of the lean loss]
The only functional critical point of $\E\mathcal L_{\mathrm{AM}}$ is the optimal control $u^*$.
\end{proposition}

\begin{proof}[Reconstruction of Appendix E.3]
For a fixed control $v$, integration of (38)--(39) and the tower property give
\begin{equation*}
 \E[\widetilde a(t;X^v)\mid X_t^v]
 =\E\left[
 \int_t^T
 \left(\nabla b(X_s^v,s)^\top
 \E[\widetilde a(s;X^v)\mid X_s^v]
 +\nabla f(X_s^v,s)
 \right)ds
 +\nabla g(X_T^v)
 \middle|X_t^v
 \right].
 \tag{204}
\end{equation*}
The full adjoint satisfies
\begin{align*}
 \E[a(t;X^v,v)\mid X_t^v]
 =\E\Bigg[\int_t^T
 &\Big(
 \nabla(b+\sigma v)(X_s^v,s)^\top
 \E[a(s;X^v,v)\mid X_s^v]\\
 &+\nabla(f+\tfrac12\|v\|^2)(X_s^v,s)
 \Big)ds+\nabla g(X_T^v)
 \Bigm|X_t^v\Bigg].
 \tag{206}
\end{align*}
The expected lean loss admits the conditional-variance decomposition
\begin{align*}
 \E\mathcal L_{\mathrm{AM}}(u)
 ={}&\E\int_0^T
 \|u(X_t^u,t)+\sigma(t)^\top
 \E[\widetilde a(t;X^u)\mid X_t^u]\|^2dt\\
 &+\E\int_0^T
 \|\sigma(t)^\top(
 \E[\widetilde a(t;X^u)\mid X_t^u]-\widetilde a(t;X^u))\|^2dt,
 \tag{207}
\end{align*}
where trajectory generation is stopped for differentiation.  If $\widehat u$ is a critical point, the first variation yields
\begin{equation*}
 \widehat u(x,t)
 =-\sigma(t)^\top
 \E[\widetilde a(t;X^{\widehat u})\mid X_t^{\widehat u}=x].
 \tag{209}
\end{equation*}
Multiplying by $\nabla\widehat u$ gives the zero-conditional-mean identity
\begin{equation*}
 \nabla\widehat u^\top\sigma^\top
 \E[\widetilde a\mid X_t]
 +\nabla\widehat u^\top\widehat u=0.
 \tag{210}
\end{equation*}
Equivalently,
\begin{equation*}
 \E\left[
 \int_t^T
 \left(
 (\nabla(\sigma\widehat u))^\top
 \E[\widetilde a(s)\mid X_s]
 +\nabla\frac12\|\widehat u\|^2
 \right)ds
 \middle|X_t
 \right]=0.
 \tag{211}
\end{equation*}
Adding (211) to the lean integral equation (204) shows that
$\E[\widetilde a(t;X^{\widehat u})\mid X_t^{\widehat u}]$ satisfies the full-adjoint integral equation (206) with $v=\widehat u$:
\begin{align*}
 \E[\widetilde a(t;X^{\widehat u})\mid X_t^{\widehat u}]
 =\E\Bigg[\int_t^T
 &\Big(
 \nabla(b+\sigma\widehat u)^\top
 \E[\widetilde a(s;X^{\widehat u})\mid X_s^{\widehat u}]\\
 &+\nabla(f+\tfrac12\|\widehat u\|^2)
 \Big)ds+\nabla g(X_T^{\widehat u})
 \Bigm|X_t^{\widehat u}\Bigg].
 \tag{212}
\end{align*}
By Proposition 8, this solution is the conditional mean of the full adjoint:
\begin{equation*}
 \E[\widetilde a(t;X^{\widehat u})\mid X_t^{\widehat u}]
 =\E[a(t;X^{\widehat u},\widehat u)\mid X_t^{\widehat u}].
 \tag{212b}
\end{equation*}
Therefore the first variation of the basic loss at $\widehat u$ is
\begin{align*}
 \frac{\delta}{\delta u}\E\mathcal L_{\mathrm{Basic-AM}}(\widehat u)(x,t)
 &=2\left[
 \widehat u(x,t)+\sigma(t)^\top
 \E[a(t;X^{\widehat u},\widehat u)\mid X_t^{\widehat u}=x]
 \right]\\
 &=2\left[
 \widehat u(x,t)+\sigma(t)^\top
 \E[\widetilde a(t;X^{\widehat u})\mid X_t^{\widehat u}=x]
 \right]=0.
 \tag{214}
\end{align*}
Thus every critical point of the lean loss is a critical point of the basic loss.  Proposition 2 then gives $\widehat u=u^*$.
\end{proof}

\begin{remark}[Dependence of Proposition 7 on Proposition 8]
The uniqueness step is essential: it identifies the conditional lean adjoint with the conditional full adjoint.  Any missing Jacobian term or insufficient regularity in Proposition 8 therefore propagates directly into the proof that the practical loss has no spurious critical points.
\end{remark}

Under the memoryless schedule, the control can be parameterized by the fine-tuned native model field.  For DDIM,

and for Flow Matching,
\begin{equation*}
 u(x,t)=

 \tag{27}
\end{equation*}
Substituting these expressions into (37) gives the paper's regression objectives in the native denoiser or velocity parameterization.  The trajectory and lean-adjoint targets are stopped, so only the fine-tuned field receives gradients.  After training, the learned target score or native field can be inserted into any sampler in the equivalent family (10--11).
