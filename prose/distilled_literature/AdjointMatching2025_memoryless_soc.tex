\section{A unified stochastic sampler}
Let $(\alpha_t,\beta_t)$ describe an affine reference flow, and let $s(x,t)$ be the score of the
intermediate distribution.  Flow-matching and diffusion samplers can be written in the common form
\begin{equation}\label{eq:AM-base}
 dX_t=b(X_t,t)dt+\sigma(t)\,dB_t,
 \qquad X_0\sim\mathcal N(0,I),
\end{equation}
with
\[
 b(x,t)=\kappa_tx+
 \left(\frac{\sigma(t)^2}{2}+\eta_t\right)s(x,t),
 \qquad
 \kappa_t=\frac{\dot\alpha_t}{\alpha_t},
 \qquad
 \eta_t=\beta_t
 \left(\frac{\dot\alpha_t}{\alpha_t}\beta_t-dot\beta_t\right).
\]
Changing $\sigma$ changes the pathwise sampler but preserves the one-time marginals generated by the
same score field.

\section{Control-affine stochastic optimal control}
For running cost $f$ and terminal cost $g$, consider
\begin{equation}\label{eq:AM-soc}
\begin{aligned}
 \inf_u\quad&
 \E\left[
 \int_0^1\left(\frac12\|u(X_t^u,t)\|^2+f(X_t^u,t)\right)dt
 +g(X_1^u)\right],\\
 \text{s.t.}\quad&
 dX_t^u=\{b(X_t^u,t)+\sigma(t)u(X_t^u,t)\}dt+\sigma(t)\,dB_t.
\end{aligned}
\end{equation}
The value function
\[
 V(x,t)=\inf_u\E\left[
 \int_t^1\left(\frac12\|u(X_s,s)\|^2+f(X_s,s)\right)ds+g(X_1)
 \mid X_t=x\right]
\]
solves the Hamilton--Jacobi--Bellman equation
\[
 \partial_tV+
 \langle b,\nabla V\rangle+\frac{\sigma^2}{2}\Delta V
 -\frac12\|\sigma\nabla V\|^2+f=0,
 \qquad V(\cdot,1)=g,
\]
and the optimal feedback is
\begin{equation}\label{eq:AM-control}
 u^*(x,t)=-\sigma(t)^\top\nabla V(x,t).
\end{equation}
The logarithmic transformation of the HJB equation gives the path-integral representation
\[
 V(x,t)=-\log\E_{\rm base}
 \left[
 \exp\left(-\int_t^1f(X_s,s)ds-g(X_1)\right)\mid X_t=x
 \right].
\]
Girsanov's theorem gives
\[
 \KL(P^u(\cdot\mid X_0)\|P^{\rm base}(\cdot\mid X_0))
 =\frac12\E_{P^u}\int_0^1\|u(X_t,t)\|^2dt.
\]
Thus \eqref{eq:AM-soc} is a path-space KL-regularized reward problem.

\section{The initial-value bias}
Conditionally on $X_0$, the optimal path law is
\[
 \frac{dP^*}{dP^{\rm base}}(X\mid X_0)
 =\frac{\exp(-\int_0^1f(X_t,t)dt-g(X_1))}
        {\exp(-V(X_0,0))}.
\]
Multiplying by the initial law shows that the unconditional optimum includes the factor
$\exp(V(X_0,0))$.  For reward fine-tuning, $f=0$ and $g=-r$, so the desired terminal law is
$p^{\rm base}_1(x)e^{r(x)}/Z$, but in general the dependence between $X_0$ and $X_1$ makes the
initial normalization bias the terminal marginal.

\section{Memoryless samplers}
\begin{definition}
A sampler of the form \eqref{eq:AM-base} is memoryless when $X_0$ and $X_1$ are independent.
\end{definition}
If the base sampler is memoryless, then
\[
 p^*(X_1)
 =\int p^{\rm base}(X_0)p^{\rm base}(X_1)
       e^{r(X_1)+V(X_0,0)}dX_0
 \propto p^{\rm base}(X_1)e^{r(X_1)}.
\]
Thus ordinary path-space KL control produces the correct exponentially tilted terminal law.

\begin{proposition}[Memoryless noise schedules]
Within the family \eqref{eq:AM-base}, the endpoint variables are independent if and only if
\begin{equation}\label{eq:AM-memoryless}
 \sigma(t)^2=2\eta_t+\chi(t)
\end{equation}
for a function $\chi$ satisfying, for every $t>0$,
\begin{equation}\label{eq:AM-limit}
 \lim_{t'\downarrow0}
 \alpha_{t'}
 \exp\left(-\int_{t'}^t\frac{\chi(s)}{2\beta_s^2}ds\right)=0.
\end{equation}
The canonical choice is $\chi=0$, namely
$\sigma(t)=\sqrt{2\eta_t}$.
\end{proposition}

\begin{proof}
Reverse time and use the equality in law between the forward score sampler and its reversed
representation.  Solving the resulting linear stochastic equation by variation of constants isolates
the coefficient multiplying the initial variable.  After substituting
$\sigma^2=2\eta+\chi$, this coefficient is exactly the limit in
\eqref{eq:AM-limit}.  If the limit is zero, all remaining terms are stochastic integrals and score
functionals independent of the reversed initial variable, so the endpoints are independent.  If the
limit is nonzero, the terminal variable retains a nonzero component of the initial variable; in
general independence fails.  This proves necessity and sufficiency.
\end{proof}

\begin{theorem}[Fine-tuning with arbitrary post-training samplers]
Suppose a score or velocity model is to be fine-tuned by the KL-control problem
\eqref{eq:AM-soc} with $f=0$ and $g=-r$, while retaining the freedom to sample afterward with any
noise schedule in the marginally equivalent family \eqref{eq:AM-base}.  The control problem must be
solved with the canonical memoryless schedule
\[
 \sigma(t)=\sqrt{2\eta_t}.
\]
With that schedule, the optimum modifies the score by
\begin{equation}\label{eq:AM-score-control}
 s^*(x,t)=s(x,t)+\frac{u^*(x,t)}{\sqrt{2\eta_t}},
\end{equation}
and the resulting score may then be inserted into any of the equivalent samplers.
\end{theorem}

\begin{proof}
Write the Fokker--Planck equations for the base and target noising processes and set their negative
log densities to $V$ and $V^*$.  Their difference $\widehat V=V^*-V$ satisfies an HJB equation.  Under
the schedule $\sigma^2=2\eta$, that HJB is exactly the value equation for
\eqref{eq:AM-soc} with terminal reward $r$.  Equation \eqref{eq:AM-control} then gives
\[
 u^*=\sqrt{2\eta_t}(s^*-s),
\]
which is \eqref{eq:AM-score-control}.  The memoryless proposition removes the initial-value bias and
gives the desired terminal tilt.  Conversely, a nonmemoryless schedule retains an endpoint
correlation and therefore cannot guarantee the same terminal tilt for arbitrary rewards while
preserving schedule conversion through the score relation.
\end{proof}

\section{The continuous adjoint}
For a fixed control $u$, let $a(t;X,u)$ be the backward adjoint along a controlled trajectory.  It
satisfies a terminal condition $a(1)=\nabla g(X_1)$ and a backward ODE containing the Jacobian of
the controlled drift and the gradient of the running cost.  The pathwise derivative of the control
objective is
\[
 \frac{dL}{d\theta}
 =\int_0^1
 \left\langle
 u(X_t,t)+\sigma(t)^\top a(t;X,u),
 \partial_\theta u(X_t,t)
 \right\rangle dt.
\]

\section{Basic Adjoint Matching}
Freeze the trajectory-generating control $\bar u=\operatorname{stopgrad}(u)$ and define
\begin{equation}\label{eq:AM-basic-loss}
 L_{\rm basic}(u;X)
 :=\frac12\int_0^1
 \|u(X_t,t)+\sigma(t)^\top a(t;X,\bar u)\|^2dt,
 \qquad X\sim P^{\bar u}.
\end{equation}
Because the trajectory and target are stop-gradient quantities, differentiating
\eqref{eq:AM-basic-loss} gives exactly the continuous-adjoint gradient above.

If $u$ is a critical point, then
\[
 u(x,t)=-\sigma(t)^\top
 \E[a(t;X,u)\mid X_t=x].
\]
The conditional expected adjoint is the spatial gradient of the cost-to-go $J(u;x,t)$.  Hence the
critical-point equation is
$u=-\sigma^\top\nabla J(u)$.  Substituting this relation into the backward Kolmogorov equation
for $J(u)$ turns it into the HJB equation.  Uniqueness of the HJB solution implies
$J(u)=V$ and therefore $u=u^*$.

\begin{proposition}
The expected basic Adjoint Matching loss has exactly one critical point, the optimal control $u^*$.
Its parameter gradient equals the continuous-adjoint gradient.
\end{proposition}

\section{Lean-adjoint regression in the paper's terminology}
The practical loss removes control-dependent terms from the target adjoint, producing a lower-
variance ``lean adjoint'' $\widetilde a$.  A conditional-expectation consistency penalty is added so
that at a critical point
\[
 u(x,t)=-\sigma(t)^\top
 \E[\widetilde a(t;X)\mid X_t=x].
\]
The tower property and this critical-point identity show that
$\E[\widetilde a\mid X_t]$ satisfies the same linear integral equation as
$\E[a(t;X,u)\mid X_t]$.  A Gronwall argument proves uniqueness of that equation.  Therefore every
critical point of the practical loss is a critical point of the basic loss, and hence equals $u^*$.
