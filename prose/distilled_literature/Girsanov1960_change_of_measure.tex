\section*{Reference and purpose}
This note expands I. V. Girsanov, ``On Transforming a Certain Class of Stochastic Processes by Absolutely Continuous Substitution of Measures,'' \emph{Theory of Probability and Its Applications}, 5(3):285--301, 1960 \cite{Girsanov1960}.  This is a non-arXiv classical source.  The notation in the translated PDF is old and the scan is noisy; this note keeps the proof architecture and rewrites it in modern stochastic-process language.

\section{Why the paper matters}
For dynamic Schrodinger bridge and stochastic optimal control, the key principle is:
\begin{quote}
Changing path measure by an exponential stochastic integral changes the drift of a Wiener or diffusion process while preserving the diffusion coefficient.
\end{quote}
This is the rigorous bridge between path-space relative entropy and quadratic control energy.  It is not needed for the current finite Sinkhorn projective contraction, but it is essential provenance for continuum SOC/bridge arguments.

\section{Wiener processes in Girsanov's notation}
Girsanov works on a filtered probability space and writes an $n$-dimensional process as
\[
  X=\{x(t,\omega),P\},\qquad 0\le t\le1.
\]
A Wiener process is characterized by conditional moment relations with respect to the past sigma-algebra $\mathcal F_s$:
\begin{align*}
  \E[x_i(t)-x_i(s)\mid\mathcal F_s]&=0,\\
  \E[(x_i(t)-x_i(s))(x_j(t)-x_j(s))\mid\mathcal F_s]&=\delta_{ij}(t-s).
\end{align*}
Girsanov notes that Levy's theorem identifies such a process with Brownian motion in the wide sense.

\section{Ito processes and coefficient assumptions}
The paper considers an Ito process driven by a Wiener process $\xi$:
\begin{equation}
  dx(t)=a(t,\omega)\,dt+q(t,\omega)\,d\xi(t). \tag{Ito}\label{eq:girs-ito}
\end{equation}
Here $q$ is an $n\times n$ matrix process and $a$ is a vector process.  The paper imposes assumptions labelled $(A_0)$--$(A_3)$, which in modern terms ensure:
\begin{enumerate}[leftmargin=2em]
\item adaptedness/progressive measurability of the coefficients;
\item enough square-integrability to define the Ito integrals;
\item nondegeneracy or boundedness conditions when required by the approximation lemmas;
\item existence and uniqueness of the diffusion process when Section 3 applies the theorem to SDEs.
\end{enumerate}
The exact historical hypotheses are stronger and more technical than modern Novikov/Kazamaki formulations; the proof proceeds by bounded truncations and limiting arguments rather than by a single martingale criterion.

\section{The exponential density}
Let $\theta(t,\omega)$ be an adapted vector process satisfying the relevant assumptions.  Girsanov defines an exponential density of the form
\begin{equation}
  Z_t=\exp\left(\int_0^t \theta(s)\cdot d\xi(s)-\frac12\int_0^t\|\theta(s)\|^2ds\right). \tag{Z}\label{eq:girs-Z}
\end{equation}
The paper writes this with its own notation $\varphi(t,\omega)$ and assumes the normalization
\begin{equation}
  \E_P Z_1=1. \tag{Norm}\label{eq:girs-norm}
\end{equation}
Then a new probability measure is defined by
\begin{equation}
  d\widetilde P = Z_1\,dP. \tag{Measure}\label{eq:girs-measure}
\end{equation}
The proof spends much of Section 2 verifying that this density is legitimate under the stated hypotheses, first in bounded cases and then by approximation.

\section{Fundamental Theorem 1}
The core theorem can be stated in modern notation as follows.

\begin{distilled}[Girsanov Theorem 1, modern form]
Let $\xi$ be an $n$-dimensional Wiener process under $P$, and let $X$ satisfy \eqref{eq:girs-ito}.  Let $\theta$ be an adapted process such that the exponential $Z$ in \eqref{eq:girs-Z} is normalized by \eqref{eq:girs-norm}.  Define $\widetilde P$ by \eqref{eq:girs-measure}.  Then
\begin{equation}
  \widetilde\xi(t):=\xi(t)-\int_0^t\theta(s)\,ds \tag{newB}\label{eq:girs-newB}
\end{equation}
is a Wiener process under $\widetilde P$.  Moreover $X$ is an Ito process under $\widetilde P$ with the same diffusion matrix $q$ and drift shifted to
\begin{equation}
  \widetilde a(t,\omega)=a(t,\omega)+q(t,\omega)\theta(t,\omega). \tag{drift}\label{eq:girs-drift}
\end{equation}
\end{distilled}

Equivalently, if under $P$
\[
  dX_t=a_tdt+q_td\xi_t,
\]
then under $\widetilde P$
\[
  dX_t=(a_t+q_t\theta_t)dt+q_td\widetilde\xi_t.
\]

\section{Proof of Theorem 1: main steps}
\subsection{Step 1: Ito formula for the exponential}
Girsanov first recalls Ito's formula for $f(X_t)$ when $X$ is an Ito process.  Applying it to $f(u)=e^u$ and
\[
  U_t=\int_0^t\theta_s\cdot d\xi_s-\frac12\int_0^t\|\theta_s\|^2ds
\]
gives the stochastic differential
\[
  dZ_t=Z_t\theta_t\cdot d\xi_t.
\]
This identifies $Z$ as a local martingale and supplies the change-of-variable formula for integrals weighted by $Z_t$.

\subsection{Step 2: verify Brownian conditional moments under the new measure}
To prove that $\widetilde\xi$ is Wiener under $\widetilde P$, it is enough, in Girsanov's framework, to verify the conditional moment identities:
\begin{align*}
  \widetilde\E[\widetilde\xi_i(t)-\widetilde\xi_i(s)\mid\mathcal F_s]&=0,\\
  \widetilde\E[(\widetilde\xi_i(t)-\widetilde\xi_i(s))(\widetilde\xi_j(t)-\widetilde\xi_j(s))\mid\mathcal F_s]&=\delta_{ij}(t-s).
\end{align*}
Because $d\widetilde P=Z_1dP$, conditional expectations under $\widetilde P$ are rewritten as weighted conditional expectations under $P$.  The stochastic differential $dZ_t=Z_t\theta_t\cdot d\xi_t$ cancels the finite-variation drift introduced by subtracting $\int\theta ds$.

\subsection{Step 3: bounded controls}
The proof first treats bounded controls, for example controls with
\[
  0<c_1\le |\theta(t,\omega)|\le c_2<\infty
\]
or with bounded truncations depending on the exact lemma.  In this case all stochastic integrals are square-integrable, the exponential is uniformly controlled enough for conditional expectations, and Fatou/dominated-convergence arguments are straightforward.

\subsection{Step 4: approximation by truncation}
The general case is obtained by truncating $\theta$.  Girsanov uses cutoff functions denoted $W_{2N}$ in the paper, replacing $\theta$ by bounded approximants
\[
  \theta^{(N)}(t,
\omega)=W_{2N}(\theta(t,\omega))\theta(t,\omega).
\]
For each $N$, the bounded theorem applies.  Lemmas 2--5 show that:
\begin{enumerate}[leftmargin=2em]
\item the exponential densities $Z^{(N)}$ converge to $Z$ in the required sense;
\item stochastic integrals with respect to the approximating measures converge;
\item the conditional moment identities pass to the limit;
\item Fatou's lemma controls the limiting integrability.
\end{enumerate}
This is why the paper assumes the normalization $\E Z_1=1$ explicitly: it is the global condition that prevents mass from escaping in the limiting measure.

\subsection{Step 5: drift transformation for Ito processes}
Once \eqref{eq:girs-newB} is Brownian under $\widetilde P$, the drift statement is algebraic:
\[
  d\xi_t=d\widetilde\xi_t+\theta_tdt.
\]
Substituting into \eqref{eq:girs-ito} gives
\[
  dX_t=a_tdt+q_t(d\widetilde\xi_t+\theta_tdt)
      =(a_t+q_t\theta_t)dt+q_td\widetilde\xi_t.
\]
The paper formulates this as $X$ being an Ito process with respect to the new Wiener process and with shifted transfer vector.

\section{Lemma structure in Section 2}
The original proof lists several lemmas.  Their modern roles are:
\begin{enumerate}[leftmargin=2em]
\item \textbf{Lemma 1: boundedness/nontriviality of the exponential density.}  Under bounded hypotheses the density is positive and finite, and the transformed measure is well-defined.
\item \textbf{Lemma 2: exponential convergence under admissible approximations.}  If the truncated controls satisfy the assumptions, the corresponding exponentials converge to the target exponential.
\item \textbf{Lemma 3: change of conditional expectations.}  Weighted conditional expectations under $P$ are rewritten as ordinary conditional expectations under $\widetilde P$.
\item \textbf{Lemma 4: integrability of the density process.}  The density process defined by the stochastic exponential has expectation one at intermediate times in the bounded case.
\item \textbf{Lemma 5: passage from bounded truncations to the general normalized case.}  If $\int e^{\Phi}\,dP=1$ and approximating controls are used, the limiting density defines the desired absolutely continuous measure.
\item \textbf{Lemma 6: stochastic-integral invariance.}  The stochastic integrals appearing in the Ito representation can be passed from the original Wiener process to the shifted Wiener process under the new measure.
\end{enumerate}
The paper's presentation is more technical because it proves these facts without the modern compact language of exponential martingales.

\section{Diffusion-type processes and Theorem 2}
In Section 3 Girsanov applies Theorem 1 to diffusion processes whose coefficients depend on the path.  A diffusion-type process has drift/transfer vector $A(t,x(\cdot))$ and diffusion matrix $B(t,x(\cdot))$.  The theorem compares two processes with the same diffusion coefficient but drifts differing by $B\theta$.

Let $X$ solve, schematically,
\[
  dX_t = \big(A(t,X)+B(t,X)\theta(t,X)\big)dt+B(t,X)dW_t.
\]
Under the density induced by the stochastic exponential associated with $\theta(t,X)$, the shifted process
\[
  \widetilde W_t=W_t-\int_0^t\theta(s,X)ds
\]
is Brownian, and $X$ has drift $A(t,X)$ under the new measure:
\[
  dX_t=A(t,X)dt+B(t,X)d\widetilde W_t.
\]
The theorem assumes uniqueness for the relevant SDE equation so that equality of laws follows from equality of coefficients under the transformed measure.

\begin{distilled}[Girsanov Theorem 2, modern form]
For diffusion-type processes with a common nondegenerate diffusion matrix $B$ and drifts differing by $B\theta$, an absolutely continuous change of path measure with density $\exp Z(X(\cdot))$ transforms one diffusion law into the other.  The Radon--Nikodym derivative between the induced path measures is the exponential functional used in Theorem 1, provided the SDE has a unique solution and the growth hypotheses ensure integrability.
\end{distilled}

\section{Entropy/energy identity used in Schrodinger bridges}
Girsanov's paper focuses on absolute continuity and drift transformation.  The identity most often used in Schrodinger bridge/SOC applications is the later standard consequence:
if $Q\ll P$ is obtained by adding drift $u$ to Brownian motion and
\[
  \frac{dQ}{dP}=\exp\left(\int_0^T u_t\cdot dW_t-rac12\int_0^T\|u_t\|^2dt\right),
\]
then, under the regularity assumptions making the stochastic integral a true martingale,
\begin{equation}
  \KL(Q\mid P)=\frac12\E_Q\int_0^T\|u_t\|^2dt. \tag{energy}\label{eq:girs-energy}
\end{equation}
This follows by taking $Q$-expectations of $\log(dQ/dP)$ and rewriting the stochastic integral under $Q$ using the shifted Brownian motion.  The martingale term has zero $Q$-expectation.  Equation \eqref{eq:girs-energy} is the analytic bridge from entropy minimization over path measures to quadratic stochastic control.

\section{Formalization relevance}
\begin{formalizationnotes}
\item For finite Sinkhorn proofs, this paper is background only.
\item For continuum bridge/SOC files, Theorem 1 is the source-level statement that drift changes are equivalent to absolutely continuous path-measure changes.
\item A direct Lean formalization would require Brownian motion, stochastic integration, exponential martingales, and SDE uniqueness.  That is far beyond ordinary measure theory.
\item For near-term formalization, use this note to justify theorem seams such as ``Girsanov change-of-drift theorem'' and ``entropy equals half control energy'' rather than attempting to prove them from first principles.
\item Be careful with signs: the new Brownian motion is either $W-\int u$ or $W+\int u$ depending on whether the density is written with $+\int u\,dW$ or $-\int u\,dW$.
\end{formalizationnotes}
